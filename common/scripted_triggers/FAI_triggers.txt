has_green_air = {
	NOT = {
		any_enemy_country = {
			has_deployed_air_force_size = {
				size > ROOT.num_deployed_planes_with_type@small_plane_airframe
				type = fighter
			}
		}
	}
}

has_red_air = {
	any_enemy_country = {
		has_deployed_air_force_size = {
			size > ROOT.num_deployed_planes_with_type@small_plane_airframe
			type = fighter
		}
	}
}

has_manpower_and_equipment_surplus = {
	manpower_per_military_factory > 1000
	stockpile_ratio = { archetype = infantry_equipment ratio > 0.20 }
	stockpile_ratio = { archetype = support_equipment ratio > 0.20 }
	stockpile_ratio = { archetype = artillery_equipment ratio > 0.20 }
}

has_manpower_or_equipment_shortage = {
	OR = {
		manpower_per_military_factory < 500
		stockpile_ratio = { archetype = infantry_equipment ratio < 0.01 }
		stockpile_ratio = { archetype = support_equipment ratio < 0.01 }
		stockpile_ratio = { archetype = artillery_equipment ratio < 0.01 }
	}
}

has_an_actual_civil_war = {
	has_civil_war = yes
	NOT = {
		any_enemy_country = {
			has_civil_war = no
		}
	}
}
